x=float(input("Digite a sua nota"))
y=float(input("Digite a sua segunda nota"))
media=(x+y)/2
if(media>=9):
    print("A")
elif(media>=7.5):
    print("B")
elif(media>=6):
    print("C")
elif(media>=4):
    print("D")
else:
    print('E')
